Radio groups compose Kobalte’s radio-group primitive with item wrappers, indicators, and optional descriptions and error messages.

## Usage

```tsx
import {
  RadioGroup,
  RadioGroupDescription,
  RadioGroupErrorMessage,
  RadioGroupItem,
  RadioGroupItemControl,
  RadioGroupItemIndicator,
  RadioGroupItemInput,
  RadioGroupItemLabel,
  RadioGroupItems,
  RadioGroupLabel,
} from "shadcn-solid-components/components/radio-group"
```

Structure each option with `RadioGroupItem`, `RadioGroupItemInput`, `RadioGroupItemControl`, `RadioGroupItemIndicator`, and `RadioGroupItemLabel` inside `RadioGroupItems`.

## Exports

- `RadioGroup`, `RadioGroupItems`, `RadioGroupItemInput`, `RadioGroupItem`, `RadioGroupItemControl`, `RadioGroupItemIndicator`, `RadioGroupLabel`, `RadioGroupItemLabel`, `RadioGroupDescription`, `RadioGroupErrorMessage`
- Types: matching `*Props` for each export

  ```tsx
  import { For } from "solid-js"

import {
  RadioGroup,
  RadioGroupItem,
  RadioGroupItemControl,
  RadioGroupItemIndicator,
  RadioGroupItemInput,
  RadioGroupItemLabel,
  RadioGroupItems,
  RadioGroupLabel,
} from "shadcn-solid-components/components/radio-group"

const RadioGroupDemo = () => {
  return (
    <RadioGroup defaultValue="Orange">
      <RadioGroupLabel>Favorite fruit</RadioGroupLabel>
      <RadioGroupItems>
        <For each={["Apple", "Orange", "Watermelon"]}>
          {(fruit) => (
            <RadioGroupItem value={fruit}>
              <RadioGroupItemInput />
              <RadioGroupItemControl>
                <RadioGroupItemIndicator />
              </RadioGroupItemControl>
              <RadioGroupItemLabel>{fruit}</RadioGroupItemLabel>
            </RadioGroupItem>
          )}
        </For>
      </RadioGroupItems>
    </RadioGroup>
  )
}

export default RadioGroupDemo
  ```
